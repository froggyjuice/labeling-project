# 환경 변수 템플릿 파일
# 이 파일을 복사하여 .env 파일로 만들고 실제 값을 입력하세요
# .env 파일은 .gitignore에 포함되어 Git에 업로드되지 않습니다

# Django 설정
DJANGO_SECRET_KEY=your-secret-key-here
DJANGO_SETTINGS_MODULE=labeling_project.settings

# 데이터베이스 (로컬 개발용)
DATABASE_URL=postgresql://username:password@localhost:5432/labeling_project

# Google OAuth (새로 발급받은 키로 교체 필요!)
GOOGLE_OAUTH_CLIENT_ID=your-google-client-id
GOOGLE_OAUTH_CLIENT_SECRET=your-google-client-secret
GOOGLE_API_KEY=your-google-api-key

# AWS S3 (선택사항)
USE_S3=FALSE
AWS_ACCESS_KEY_ID=your-aws-access-key
AWS_SECRET_ACCESS_KEY=your-aws-secret-key
AWS_STORAGE_BUCKET_NAME=your-bucket-name

# 기타
DEBUG=True
ADMIN_EMAIL=admin@example.com

# 사용법:
# 1. 이 파일을 .env로 복사: cp env_template.txt .env
# 2. .env 파일에 실제 API 키 값들을 입력
# 3. Django에서 python-dotenv로 자동 로드
